\section{多体系の統計力学}

\begin{frame}[t,fragile]{典型的な統計力学モデル}
  \begin{itemize}
    %\setlength{\itemsep}{1em}
  \item 古典粒子系
    \begin{itemize}
    \item 調和振動子 \ \ $\displaystyle H = \frac{p^2}{2m} + \frac{k}{2}x^2$
    \item 多粒子系
      \[
      H = \sum \frac{p_i^2}{2m} + \sum_{ij} V(x_i, x_j)
      \]
    \item バネビーズ模型
      \[
      H = \sum \frac{p_i^2}{2m} + \frac{k}{2} \sum_{ij} (x_i-x_j)^2
      \]
  \end{itemize}
  \item 磁性体
    \begin{itemize}
    \item イジング模型 \ \ $\displaystyle H = -J \sum_{ij} \sigma_i \sigma_j$
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}[t,fragile]{多体系の統計力学}
  \begin{itemize}
    \setlength{\itemsep}{1em}
  \item カノニカル分布 \ $P(c) = \exp [- H(c) / k_{\rm B}T ] / Z$
  \item 分配関数・自由エネルギー
    \begin{align}
      Z(T) &= \int \exp [- H(p,q) / k_{\rm B}T ] \, dp \, dq \\
      f(T) = - k_{\rm B} T \log Z(T)
    \end{align}
  \item 物理量の期待値
  \end{itemize}
\end{frame}
