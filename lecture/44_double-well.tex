\section{二重井戸ポテンシャル}

\begin{frame}[t,fragile]{二重井戸ポテンシャル中の粒子}
  \begin{itemize}
    %\setlength{\itemsep}{1em}
  \item 時間依存しないシュレディンガー方程式
    \begin{align*}
      \big[ -\frac{d^2}{dx^2} + V(x) \big] \psi(x) = E \psi(x)
    \end{align*}
    ($\hbar^2/2m = 1$となるように単位をとった)
  \item 二重井戸ポテンシャル
    \begin{align*}
      V(x) = \begin{cases}
        \infty & \text{$x < 0$, $x > 1$} \\
        0 & \text{$0 < x < a$, $b < x < 1$} \\
        v & \text{$a < x < b$}
      \end{cases}
    \end{align*}
    ただし、$0<a<b<1$とする
  \item 境界条件: $\psi(0) = \psi(1) = 0$、$0 < x < 1$で$\psi(x)$とその導関数が連続
  \end{itemize}
\end{frame}
