  \item 2サイト系
    \[
      H = -J \sigma_1^z \sigma_2^z - h (\sigma_1^z + \sigma_2^z) - \Gamma (\sigma_1^x + \sigma_2^x)

      \langle \uparrow \uparrow \!| H |\! \uparrow \uparrow \rangle &= -J - 2h \\

      \langle \uparrow \uparrow \!| H |\! \downarrow \downarrow \rangle &= 0 \\
      &\vdots
    \end{align*}
